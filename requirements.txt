# OpenClaw VPS 部署依赖清单
# 安装: pip install -r requirements.txt
#
# 目录结构:
#   longbridge/ - 长桥 SDK 封装核心服务
#   strategies/ - 量化交易策略 (盘前谋划 + 盘中哨兵)
#   notion/     - Notion 账簿管理 (MCP 服务 + TG 指令机器人 + 价格同步器)
#   prompts/    - AI 系统提示词
#   data/       - 运行时数据 (交易计划 JSON + 缓存)

# ──────────────────────────────────────────
# 🔧 核心基础
# ──────────────────────────────────────────
python-dotenv           # 读取 .env 环境变量
pytz                    # 时区处理（港/美股交易时段判断）
requests                # HTTP 请求（TG 推送、Tavily 新闻）

# ──────────────────────────────────────────
# 📈 长桥 SDK（core/ + strategies/）
# ──────────────────────────────────────────
longbridge              # 长桥官方 Python SDK（行情 + 交易）
pandas                  # 数据处理（YahooQuery 期权降级通道需要）
yahooquery              # 美股期权行情降级通道（长桥无权限时自动切换）

# ──────────────────────────────────────────
# 🤖 AI 服务集成（strategies/）
# ──────────────────────────────────────────
schedule                # 定时任务（premarket_planner 定时触发）

# ──────────────────────────────────────────
# 🗃️ Notion 账簿（notion/）
# ──────────────────────────────────────────
notion-client           # Notion 官方 Python SDK
python-telegram-bot     # Telegram Bot SDK（tg_trade_bot 交易指令机器人）
fastmcp                 # MCP 服务框架（notion_mcp_server）
apscheduler             # 定时任务调度（notion_price_syncer 用，比 schedule 更强大）
